function merge(arr1, arr2) {
let i = 0, j = 0, k = 0;
let result = [];
// 步骤1: 比较并合并
while (i < arr1.length && j < arr2.length) {
if (arr1[i] <= arr2[j]) {
result[k] = arr1[i];
i++;
} else {
result[k] = arr2[j];
j++;
}
k++;
}
// 步骤2: 复制剩余元素
while (i < arr1.length) {
result[k] = arr1[i];
i++; k++;
}
while (j < arr2.length) {
result[k] = arr2[j];
j++; k++;
}
return result;
}
