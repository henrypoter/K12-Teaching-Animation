        function findMaxGrowthIndex(history) {
            let maxGrowth = 0;
            let maxIndex = 0;
            
            for (let i = 1; i < history.length; i++) {
                const growth = history[i] - history[i-1];
                if (growth > maxGrowth) {
                    maxGrowth = growth;
                    maxIndex = i;
                }
            }
            
            return maxIndex;
        }

            });
